25pub fn read_and_decrypt<T: Read>(
26    decryptor: &Arc<dyn BlockDecryptor>,
27    input: &mut T,
28    aad: &[u8],
29) -> Result<Vec<u8>> {
30    let mut len_bytes = [0; 4];
31    input.read_exact(&mut len_bytes)?;
32    let ciphertext_len = u32::from_le_bytes(len_bytes) as usize;
33    let mut ciphertext = vec![0; 4 + ciphertext_len];
34    input.read_exact(&mut ciphertext[4..])?;
35
36    decryptor.decrypt(&ciphertext, aad.as_ref())
37}

39// CryptoContext is a data structure that holds the context required to
40// decrypt parquet modules (data pages, dictionary pages, etc.).
41#[derive(Debug, Clone)]
42pub(crate) struct CryptoContext {
43    pub(crate) row_group_idx: usize,
44    pub(crate) column_ordinal: usize,
45    pub(crate) page_ordinal: Option<usize>,
46    pub(crate) dictionary_page: bool,
47    // We have separate data and metadata decryptors because
48    // in GCM CTR mode, the metadata and data pages use
49    // different algorithms.
50    data_decryptor: Arc<dyn BlockDecryptor>,
51    metadata_decryptor: Arc<dyn BlockDecryptor>,
52    file_aad: Vec<u8>,
53}
54
55impl CryptoContext {
56    pub(crate) fn new(
57        row_group_idx: usize,
58        column_ordinal: usize,
59        data_decryptor: Arc<dyn BlockDecryptor>,
60        metadata_decryptor: Arc<dyn BlockDecryptor>,
61        file_aad: Vec<u8>,
62    ) -> Self {
63        Self {
64            row_group_idx,
65            column_ordinal,
66            page_ordinal: None,
67            dictionary_page: false,
68            data_decryptor,
69            metadata_decryptor,
70            file_aad,
71        }
72    }
73
74    pub(crate) fn with_page_ordinal(&self, page_ordinal: usize) -> Self {
75        Self {
76            row_group_idx: self.row_group_idx,
77            column_ordinal: self.column_ordinal,
78            page_ordinal: Some(page_ordinal),
79            dictionary_page: false,
80            data_decryptor: self.data_decryptor.clone(),
81            metadata_decryptor: self.metadata_decryptor.clone(),
82            file_aad: self.file_aad.clone(),
83        }
84    }
85
86    pub(crate) fn create_page_header_aad(&self) -> Result<Vec<u8>> {
87        let module_type = if self.dictionary_page {
88            ModuleType::DictionaryPageHeader
89        } else {
90            ModuleType::DataPageHeader
91        };
92
93        create_module_aad(
94            self.file_aad(),
95            module_type,
96            self.row_group_idx,
97            self.column_ordinal,
98            self.page_ordinal,
99        )
100    }
101
102    pub(crate) fn create_page_aad(&self) -> Result<Vec<u8>> {
103        let module_type = if self.dictionary_page {
104            ModuleType::DictionaryPage
105        } else {
106            ModuleType::DataPage
107        };
108
109        create_module_aad(
110            self.file_aad(),
111            module_type,
112            self.row_group_idx,
113            self.column_ordinal,
114            self.page_ordinal,
115        )
116    }
117
118    pub(crate) fn for_dictionary_page(&self) -> Self {
119        Self {
120            row_group_idx: self.row_group_idx,
121            column_ordinal: self.column_ordinal,
122            page_ordinal: self.page_ordinal,
123            dictionary_page: true,
124            data_decryptor: self.data_decryptor.clone(),
125            metadata_decryptor: self.metadata_decryptor.clone(),
126            file_aad: self.file_aad.clone(),
127        }
128    }
129
130    pub(crate) fn data_decryptor(&self) -> &Arc<dyn BlockDecryptor> {
131        &self.data_decryptor
132    }
133
134    pub(crate) fn file_aad(&self) -> &Vec<u8> {
135        &self.file_aad
136    }
137}

139/// FileDecryptionProperties hold keys and AAD data required to decrypt a Parquet file.
140#[derive(Debug, Clone, PartialEq)]
141pub struct FileDecryptionProperties {
142    footer_key: Vec<u8>,
143    column_keys: HashMap<String, Vec<u8>>,
144    pub(crate) aad_prefix: Option<Vec<u8>>,
145}
146
147impl FileDecryptionProperties {
148    /// Returns a new FileDecryptionProperties builder
149    pub fn builder(footer_key: Vec<u8>) -> DecryptionPropertiesBuilder {
150        DecryptionPropertiesBuilder::new(footer_key)
151    }
152}
153
154pub struct DecryptionPropertiesBuilder {
155    footer_key: Vec<u8>,
156    column_keys: HashMap<String, Vec<u8>>,
157    aad_prefix: Option<Vec<u8>>,
158}
159
160impl DecryptionPropertiesBuilder {
161    pub fn new(footer_key: Vec<u8>) -> DecryptionPropertiesBuilder {
162        Self {
163            footer_key,
164            column_keys: HashMap::default(),
165            aad_prefix: None,
166        }
167    }
168
169    pub fn build(self) -> Result<FileDecryptionProperties> {
170        Ok(FileDecryptionProperties {
171            footer_key: self.footer_key,
172            column_keys: self.column_keys,
173            aad_prefix: self.aad_prefix,
174        })
175    }
176
177    pub fn with_aad_prefix(mut self, value: Vec<u8>) -> Self {
178        self.aad_prefix = Some(value);
179        self
180    }
181
182    pub fn with_column_key(mut self, column_name: &str, decryption_key: Vec<u8>) -> Self {
183        self.column_keys
184            .insert(column_name.to_string(), decryption_key);
185        self
186    }
187}

189#[derive(Clone, Debug)]
190pub(crate) struct FileDecryptor {
191    decryption_properties: FileDecryptionProperties,
192    footer_decryptor: Option<Arc<dyn BlockDecryptor>>,
193    file_aad: Vec<u8>,
194}
195
196impl PartialEq for FileDecryptor {
197    fn eq(&self, other: &Self) -> bool {
198        self.decryption_properties == other.decryption_properties
199    }
200}
201
202impl FileDecryptor {
203    pub(crate) fn new(
204        decryption_properties: &FileDecryptionProperties,
205        aad_file_unique: Vec<u8>,
206        aad_prefix: Vec<u8>,
207    ) -> Result<Self, ParquetError> {
208        let file_aad = [aad_prefix.as_slice(), aad_file_unique.as_slice()].concat();
209        // todo decr: if no key available yet (not set in properties, should be retrieved from metadata)
210        let footer_decryptor = RingGcmBlockDecryptor::new(&decryption_properties.footer_key)
211            .map_err(|e| {
212                general_err!(
213                    "Invalid footer key. {}",
214                    e.to_string().replace("Parquet error: ", "")
215                )
216            })?;
217        Ok(Self {
218            footer_decryptor: Some(Arc::new(footer_decryptor)),
219            decryption_properties: decryption_properties.clone(),
220            file_aad,
221        })
222    }
223
224    pub(crate) fn get_footer_decryptor(&self) -> Result<Arc<dyn BlockDecryptor>, ParquetError> {
225        Ok(self.footer_decryptor.clone().unwrap())
226    }
227
228    pub(crate) fn get_column_data_decryptor(
229        &self,
230        column_name: &str,
231    ) -> Result<Arc<dyn BlockDecryptor>, ParquetError> {
232        match self.decryption_properties.column_keys.get(column_name) {
233            Some(column_key) => Ok(Arc::new(RingGcmBlockDecryptor::new(column_key)?)),
234            None => self.get_footer_decryptor(),
235        }
236    }
237
238    pub(crate) fn get_column_metadata_decryptor(
239        &self,
240        column_name: &str,
241    ) -> Result<Arc<dyn BlockDecryptor>, ParquetError> {
242        // Once GCM CTR mode is implemented, data and metadata decryptors may be different
243        self.get_column_data_decryptor(column_name)
244    }
245
246    pub(crate) fn file_aad(&self) -> &Vec<u8> {
247        &self.file_aad
248    }
249
250    pub(crate) fn is_column_encrypted(&self, column_name: &str) -> bool {
251        // Column is encrypted if either uniform encryption is used or an encryption key is set for the column
252        match self.decryption_properties.column_keys.is_empty() {
253            false => self
254                .decryption_properties
255                .column_keys
256                .contains_key(column_name),
257            true => true,
258        }
259    }
260}
